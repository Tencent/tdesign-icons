# TDesign Icons for React

## 安装

- `npm i tdesign-icons-react`

## 使用

```js
// svgsprite
import { Icon } from "tdesign-icons-react";

<Icon name="close" />;

// iconfont
import { IconFont } from "tdesign-icons-react";

<IconFont name="close" />;

// single icon
import { CloseIcon, TimeIcon } from "tdesign-icons-react";

<CloseIcon />;
```
